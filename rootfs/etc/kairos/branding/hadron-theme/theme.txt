# Hadron Desktop GRUB theme — Tokyo Night.
#
# gfxterm theme matching the ly login + waybar palette. Used by the installed
# system's boot menu (grubmenu.cfg switches to gfxterm and `set theme=` to this
# file, which the 09_hadron_grub_theme oem stage places on COS_STATE alongside
# background.tga). desktop-image is resolved relative to this file, so the bare
# filename works once the whole hadron-theme dir lands in themes/hadron/.
#
# Image is TGA, not PNG: GRUB's png.mod rejects valid PNGs with "unsupported
# format", which makes gfxmenu discard the entire theme; tga.mod decodes cleanly.
# Font: the installed EFI GRUB ships no unicode.pf2, so grubmenu.cfg loadfont's our
# vendored unicode.pf2 (registered as "Unifont Regular 16") from STATE first.

desktop-image: "background.tga"
title-text: ""
terminal-font: "Unifont Regular 16"

    text  = "H A D R O N   D E S K T O P"
}
